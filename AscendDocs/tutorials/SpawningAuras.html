    A static class called AuraUtility has been created to facilitate easy generation of auras.

    This doc is unfinished, but for now I will add some example code for generating a circle beacon aura:

    // Spawn circle aura with linear upward movement
AutoMergeAura aura = AuraUtility.Shape.SpawnCircleAura(AuraRadius, Position + SpawnOffset, transform.parent, AuraUtility.DischargeMultipleAuraProperties);
aura.isIndicator = true;
aura.indicatorDuration = 1.0f;
BasicAuraProjectileMovement.AddToAuraObject(aura.gameObject, ProjectileProperties);

MultiEaseMovement.MultiEaseProperties ease_props = new MultiEaseMovement.MultiEaseProperties();
ease_props.Loop = false;
ease_props.KillAuraOnFinish = false;
ease_props.StartRadius = 5;
var endpoint = new MultiEaseMovement.EaseEndpoint();
endpoint.Duration = aura.indicatorDuration;
endpoint.Function = EasingFunction.Ease.Spring;
endpoint.Radius = AuraRadius;
endpoint.Freeze = false;
ease_props.EaseEndpoints = new List<MultiEaseMovement.EaseEndpoint>{ endpoint };

MultiEaseMovement.AddToAuraObject(aura.gameObject, ease_props);
